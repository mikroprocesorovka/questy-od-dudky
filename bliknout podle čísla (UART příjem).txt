#include "stm8s.h"
#include "milis.h"
//#include "stdio.h"

//#include "stm8_hd44780.h"
char input;
uint16_t previous=0;
uint16_t time=0;

void main(void){
CLK_HSIPrescalerConfig(CLK_PRESCALER_HSIDIV1); // taktovat MCU na 16MHz
init_milis(); // spustit časovač millis
GPIO_Init(GPIOC,GPIO_PIN_1,GPIO_MODE_OUT_PP_LOW_SLOW); 
UART1_Init(115200,UART1_WORDLENGTH_8D,UART1_STOPBITS_1,UART1_PARITY_NO,UART1_SYNCMODE_CLOCK_DISABLE,UART1_MODE_RX_ENABLE);
UART1_Cmd(ENABLE);
GPIO_WriteHigh(GPIOC,GPIO_PIN_1);

  while (1){
	time=milis();
	if(UART1_GetFlagStatus(UART1_FLAG_RXNE) != RESET){
		input=UART1_ReceiveData8();
		if(input>9){input=9;};
		input=input*2;	
	}
	if(time-previous>500&&input>0){
		GPIO_WriteReverse(GPIOC,GPIO_PIN_1);
		input=input-1;
		previous=time;
	}
}
}

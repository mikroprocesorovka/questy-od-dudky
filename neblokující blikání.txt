uint16_t time=0;
uint16_t previous1=0;
uint16_t previous2=0;
uint16_t previous3=0;
void blink1(void);
void blink2(void);
void blink3(void);


void main(void){
CLK_HSIPrescalerConfig(CLK_PRESCALER_HSIDIV1); // taktovat MCU na 16MHz
init_milis(); // rozběhnout milis časovač
GPIO_Init (GPIOC,GPIO_PIN_ALL,GPIO_MODE_OUT_PP_LOW_SLOW);

GPIO_WriteHigh(GPIOC,GPIO_PIN_ALL);
while (1){
	time=milis();
	blink1();
	blink2();
	blink3();
}
}
void blink1(void){
	static uint16_t previous=0;
	if((time-previous)>(1000/0.5)){
		previous=time;
		GPIO_WriteReverse(GPIOC,GPIO_PIN_1);
	}	
}	
void blink2(void){
	static uint16_t previous=0;
	if((time-previous)>(1000/1.5)){
		previous=time;
		GPIO_WriteReverse(GPIOC,GPIO_PIN_2);
	}	
}	
void blink3(void){
	static uint16_t previous=0;
	if((time-previous)>(1000/4)){
		previous=time;
		GPIO_WriteReverse(GPIOC,GPIO_PIN_3);
	}	
}

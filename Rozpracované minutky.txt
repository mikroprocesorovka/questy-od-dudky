 INTERRUPT_HANDLER(TIM3_UPD_OVF_BRK_IRQHandler, 15)
 {
	 TIM3_ClearITPendingBit(TIM3_IT_UPDATE);
	 process_enc();
 }

void init_timer(void){
TIM3_TimeBaseInit(TIM3_PRESCALER_16,1999);
TIM3_ITConfig(TIM3_IT_UPDATE, ENABLE);
TIM3_Cmd(ENABLE);
}

void init_enc(void){
GPIO_Init(GPIOE,GPIO_PIN_5,GPIO_MODE_IN_PU_NO_IT); //enkodér
GPIO_Init(GPIOC,GPIO_PIN_6,GPIO_MODE_IN_PU_NO_IT); //enkodér
GPIO_Init(GPIOC,GPIO_PIN_7,GPIO_MODE_IN_PU_NO_IT); //tlačítko enkodéru
}

void print(void){
	lcd_gotoxy(x,1);
	lcd_putchar(0);
}

void process_enc(void){
	static minule=1;
 	if(GPIO_ReadInputPin(GPIOC,GPIO_PIN_6) == RESET && minule==1){
		minule = 0; // nyní je pin v log.0
		// poeeteme stav vstupu B
		lcd_gotoxy(x,1);
		lcd_puts(" ");
		if(GPIO_ReadInputPin(GPIOE,GPIO_PIN_5) == RESET){
			x++;
			if(x>15){
				x=15;
			}
		}
		else{
			x--;
			if(x<0){
				x=0;	
			}
		}
		print();
	}
	if(GPIO_ReadInputPin(GPIOC,GPIO_PIN_6) != RESET){
	minule = 1;
	} // pokud je vstup A v log.1
	if(GPIO_ReadInputPin(GPIOC,GPIO_PIN_7) == RESET){
		hodnota++;
		if(hodnota>9){
			hodnota=0;
		}
	}
}

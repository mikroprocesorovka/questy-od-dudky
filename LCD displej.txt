#include "stm8s.h"
#include "milis.h"
#include "stdio.h"
#include "spse_stm8.h"
#include "stm8_hd44780.h"

uint8_t textove_pole[16];
uint8_t x=1;


void main(void){
CLK_HSIPrescalerConfig(CLK_PRESCALER_HSIDIV1); // taktovat MCU na 16MHz
init_milis(); // rozběhnout milis časovač
lcd_init();

	
while (1){

	lcd_gotoxy(3,0);//stará se zobrazení textu a symbolů na displej.
	lcd_puts(":");
	
	lcd_gotoxy(15,0);
	lcd_puts(".");
	
	lcd_gotoxy(8,0);
	lcd_puts("55");

	lcd_gotoxy(6,1);
	lcd_puts(textove_pole);
	sprintf(textove_pole,"%u",x);
	


}
}

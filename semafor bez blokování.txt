#include "stm8s.h"
#include "milis.h"
//#include "stdio.h"
//#include "spse_stm8.h"
//#include "stm8_hd44780.h"
uint16_t time=0;
uint16_t previoustime=0;
uint16_t antirepeat=0;




void main(void){
CLK_HSIPrescalerConfig(CLK_PRESCALER_HSIDIV1); // taktovat MCU na 16MHz
init_milis(); // rozběhnout milis časovač
GPIO_Init (GPIOC,GPIO_PIN_ALL,GPIO_MODE_OUT_PP_LOW_SLOW);
GPIO_Init (GPIOE,GPIO_PIN_4,GPIO_MODE_IN_FL_NO_IT);
GPIO_WriteHigh(GPIOC,GPIO_PIN_ALL);
while (1){
	time=milis();
	if(antirepeat==0){
		if(time-previoustime==10000){
			GPIO_WriteLow(GPIOC,GPIO_PIN_1);
			GPIO_WriteHigh(GPIOC,GPIO_PIN_2);
		}
		if((time-previoustime>=10000)&&(time-previoustime)%500==0){
			GPIO_WriteReverse(GPIOC,GPIO_PIN_1);
			GPIO_WriteReverse(GPIOC,GPIO_PIN_2);		
		}
		if(time-previoustime>=15000){
			previoustime=time;
			GPIO_WriteHigh(GPIOC,GPIO_PIN_1);
			GPIO_WriteHigh(GPIOC,GPIO_PIN_2);
		}
		antirepeat=time;
	}
	if (antirepeat!=0&&antirepeat!=time){
		antirepeat=0;
	}
	
}
}
